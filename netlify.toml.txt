# Netlify 배포를 위한 기본 설정 파일입니다.
# 이 파일은 프로젝트의 루트 디렉토리에 위치해야 합니다.

[build]
  # 웹사이트로 게시할 폴더를 'public'으로 지정합니다.
  publish = "public"
  # 서버리스 함수(api.js)가 위치한 폴더를 지정합니다.
  functions = "netlify/functions"

# Single Page Application(SPA) 및 함수 라우팅을 위한 리다이렉트 규칙입니다.
[[redirects]]
  # /api/ 로 시작하는 모든 요청은 api 함수로 전달합니다.
  from = "/api/*"
  to = "/.netlify/functions/api/:splat"
  status = 200

[[redirects]]
  # 위 규칙에 해당하지 않는 모든 요청은 index.html로 보내서
  # 클라이언트 사이드 라우팅이 작동하도록 합니다.
  from = "/*"
  to = "/index.html"
  status = 200
```
